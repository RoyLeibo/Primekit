Future<void> loadForUser(String userId) async {
  assert(
    _provider != null,
    'Call RbacService.instance.configure() before loadForUser()',
  );
  try {
    _context = await _provider!.loadContext(userId: userId);
    _isLoaded = true;
    notifyListeners();
  } catch (error) {
    throw Exception('RbacService.loadForUser failed: $error');
  }
}
