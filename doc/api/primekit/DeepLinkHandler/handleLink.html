Handles an incoming uri by finding a matching DeepLinkRoute and navigating via GoRouter.go.

Emits the URI on incomingLinks before attempting navigation so listeners can react regardless of route matching.

Does nothing if configure has not been called yet.

Implementation

void handleLink(Uri uri) {
  _incomingLinksController.add(uri);

  final router = _router;
  if (router == null) return;

  for (final route in _routes) {
    if (route.matches(uri)) {
      try {
        final path = route.handler(uri);
        router.go(path);
      } on Exception {
        // Handler threw; skip this route and try the next.
      }
      return;
    }
  }
  // No route matched; navigation is not attempted.
}
