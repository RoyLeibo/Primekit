Future<void> enqueue(EmailMessage message) async {
  await initialize();

  final item = QueuedEmail(
    id: _generateId(),
    message: message,
    enqueuedAt: DateTime.now().toUtc(),
  );

  _queue = [..._queue, item];
  await _saveToPrefs();

  _eventController.add(EmailEnqueued(item));
  PrimekitLogger.debug(
    'Enqueued email id=${item.id} to=${message.to}',
    tag: _tag,
  );
}
